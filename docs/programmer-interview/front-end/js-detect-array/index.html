Array.isArray(obj)

兼容旧版本：

if (typeof Array.isArray === 'undefined') {
    Array.isArray = function(obj) {
        return Object.prototype.toString.call(obj) === '[object Array]';
    }
}

constructor

function isArray(obj) {
    return !!obj && obj.constructor === Array;
}
