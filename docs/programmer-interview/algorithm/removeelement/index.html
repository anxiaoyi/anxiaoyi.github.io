public class RemoveElement {

    // [1,1,1,2,2]
    public int removeElement(int[] nums, int val) {
        int valid = -1;

        for (int i = 0; i < nums.length; i++) {
            if (nums[i] != val) {
                nums[++valid] = nums[i];
            }
        }

        return nums.length == 0 ? 0 : valid + 1;
    }
