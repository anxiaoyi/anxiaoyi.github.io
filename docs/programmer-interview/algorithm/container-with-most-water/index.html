public class ContainerWithMostWater {

    public int maxArea(int[] height) {
        int maxArea = Integer.MIN_VALUE;

        int lo = 0;
        int hi = height.length - 1;

        // O(n)
        while (lo < hi) {
            maxArea = Math.max(maxArea, Math.min(height[lo], height[hi]) * (hi - lo));

            // =======================================
            // æ­¤å¤„è¿™ä¸ªåœ°æ–¹ï¼Œå¿…é¡»æ˜¯å°çš„ä¸€è¾¹ç§»åŠ¨
            // å› ä¸ºå¤§çš„ç§»åŠ¨çš„è¯ï¼Œé¢ç§¯ä¸€å®šå˜å° (å®½åº¦å˜å°ï¼Œè€Œä¸”é«˜åº¦ä¸ä¼šè¶…è¿‡å°çš„)
            // è€Œå°çš„ç§»åŠ¨æœ‰å¯èƒ½å˜å¤§

            // å¦å¤–ä¸€ç§è§£é‡Šï¼š
            // æˆ‘ä»¬é€‰æ‹©ä¸€ä¸ªé«˜çš„ï¼Œä»¥ä¾¿å®¹çº³æ›´å¤šçš„æ°´
            // https://leetcode.com/problems/container-with-most-water/discuss/6100/Simple-and-clear-proofexplanation
            // =======================================
            if (height[lo] < height[hi]) {
                lo++;
            } else {
                hi--;
            }
        }

        return maxArea;
    }
