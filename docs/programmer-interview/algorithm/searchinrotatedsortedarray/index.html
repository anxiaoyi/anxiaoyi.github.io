    public boolean search0(int[] nums, int target) {
        if (nums.length == 0) {
            return false;
        }

        // ==========================
        // 1. we find the array's rotate pivot
        //
        // 4 5 6 0 0 0 1 2 3
        //       â†‘
        //     minIndex
        // ==========================
        int minIndex = searchMin(nums);

        // ==========================
        // 2. search the second half by using binary search
        //
        // 4 5 6 0 0 0 1 2 3
        //       â†‘         â†‘
        // ==========================
        if (binarySearch(nums, minIndex, nums.length - 1, target) != -1) {
            return true;
        }

        // ==========================
        // 2. search the first half by using binary search
        //
        // 4 5 6 0 0 0 1 2 3
        // â†‘   â†‘
        // ==========================
        return binarySearch(nums, 0, minIndex - 1, target) != -1;
    }

    private int searchMin(int[] nums) {
        int lo = 0; // always point to the first half
        int hi = nums.length - 1; // always point to the second half

        // ==========================
        // 1 2 3 4 5 6 7
        // â†‘           â†‘
        // ==========================
        if (nums[lo] >= nums[hi]) {
            while (lo < hi) {
                // ==========================
                // 4 5 6 1 2 3
                //     â†‘ â†‘
                //    lo hi
                // ==========================
                if (hi - lo == 1) {
                    return hi;
                }

                int m = lo + ((hi - lo) >> 1);

                // ==========================
                // we has no way to know the min located in which half
                // ==========================
                if (nums[m] == nums[lo] && nums[m] == nums[hi]) {
                    return findMin(nums, lo, hi);
                } else if (nums[m] < nums[lo]) {
                    hi = m;
                } else if (nums[m] > nums[hi]) {
                    lo = m;
                }
            }
        }

        return 0;
    }

    private int findMin(int[] nums, int lo, int hi) {
        int min = hi;
        // ==========================
        // 1 3 1 1 1 
        //     â†‘
        //    minIndex (we need return this)
        // ==========================
        for (int i = hi - 1; i >= lo; i--) {
            if (nums[i] <= nums[min]) {
                min = i;
            } else {
                break;
            }
        }

        return min;
    }

    private int binarySearch(int[] nums, int lo, int hi, int target) {
        while (lo <= hi) {
            int m = lo + ((hi - lo) >> 1);

            if (nums[m] == target) {
                return m;
            } else if (nums[m] > target) {
                hi = m - 1;
            } else {
                lo = m + 1;
            }
        }

        return -1;
    }
