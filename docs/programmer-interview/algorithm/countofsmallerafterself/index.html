import java.util.List;

// Input: [5,2,6,1]
// Output: [2,1,1,0]

public class CountOfSmallerAfterSelf {

    public List<Integer> countSmaller(int[] nums) {
        List<Integer> result = new ArrayList<>();

        for (int i = 0; i < nums.length; i++) {
            
            int curr = nums[i];
            int count = 0;

            for (int j = i + 1; j < nums.length; j++) {
                if (nums[j] < curr) {
                    count++;
                }
            }

            result.add(count);
        }

        return result;
    }

}
