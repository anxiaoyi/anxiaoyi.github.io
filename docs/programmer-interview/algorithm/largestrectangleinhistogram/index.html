柱状图中最大的矩形

描述

题目来源

给定 n 个非负整数，用来表示柱状图中各个柱子的高度。每个柱子彼此相邻，且宽度为 1 。求在该柱状图中，能够勾勒出来的矩形的最大面积。

题解

// https://leetcode.com/problems/largest-rectangle-in-histogram/ 直方图
//           __
//        __|  |
//       |  |  |
//       |  |  |   __
//  __   |  |  |__|  |
// |  |__|  |  |  |  |
// |__|__|__|__|__|__|
//
public class LargestRectangleinHistogram {

    public int largestRectangleArea(int[] heights) {
        // =================================
        // 左边最多延展到哪个索引
        // =================================

        // 左边比自己大的或相等的
        //
        // 单调栈找到第一个比自己大的或者小的数字
        int[] left = new int[heights.length];
        for (int i = 0; i < heights.length; i++) {
            int j = i;

            while (j - 1 >= 0 && heights[j - 1] >= heights[i]) {
                j--;
            }

            left[i] = j;
        }

        // =================================
        // 右边最多延展到哪个索引
        // =================================

        int[] right = new int[heights.length];
        for (int i = heights.length - 1; i >= 0; i--) {
            int j = i;

            while (j + 1 < heights.length && heights[i] <= heights[j + 1]) {
                j++;
            }

            right[i] = j;
        }

        int largest = 0;
        for (int i = 0; i < heights.length; i++) {
            largest = Math.max(heights[i] * (right[i] - left[i] + 1), largest);
        }

        return largest;
    }

    // 2, 1, 5, 6, 2, 3
    public int largestRectangleArea0(int[] heights) {
        // =================================
        // 左边最多延展到哪个索引
        // =================================

        // 左边比自己大的或相等的
        //
        // 单调栈找到第一个比自己大的或者小的数字
        Stack<Integer> stack = new Stack<>();

//           __
//        __|  |
//       |  |  |
//       |  |  |   __
//  __   |  |  |__|  |
// |  |__|  |  |  |  |
// |__|__|__|__|__|__|
//  0   1  2  3  4  5
//               ↑
// 假设当前指向的是这个
// 只要栈里面的比自身高，那么就一直 pop
// 所以 2、3 都被 pop 出去了

        int[] left = new int[heights.length];
        for (int i = 0; i < heights.length; i++) {
            int j = i;

            while (!stack.isEmpty() && heights[i] <= heights[stack.peek()]) {
                j = stack.pop();
            }

            // 栈空了，说明左边的都大于等于自己
            // 栈不空，说明栈里面的 peek() 是小于自己的，那么 peek() + 1 是大于等于自己的
            left[i] = stack.isEmpty() ? 0 : stack.peek() + 1;
            stack.push(i);
        }

        // =================================
        // 右边最多延展到哪个索引
        // =================================
        stack.clear();

        // 2,1,5,6,2,3
        //           ↑
        // 5, 5
        // 5, 4

        int[] right = new int[heights.length];
        for (int i = heights.length - 1; i >= 0; i--) {
            int j = i;

            while (!stack.isEmpty() && heights[i] <= heights[stack.peek()]) {
                j = stack.pop();
            }

            // 栈空了，说明右边的都大于等于自己
            // 栈不空，说明栈里面的 peek() 是小于自己的，那么 peek() - 1 是大于等于自己的
            right[i] = stack.isEmpty() ? heights.length - 1 : stack.peek() - 1;
            stack.push(i);
        }

        int largest = 0;
        for (int i = 0; i < heights.length; i++) {
            largest = Math.max(heights[i] * (right[i] - left[i] + 1), largest);
        }

        return largest;
    }

}
