Binary Search 二分搜索

Binary Search

// https://leetcode.com/problems/binary-search/
//
public class BinarySearch {

    public int search(int[] nums, int target) {
        if (nums == null || nums.length == 0) {
            return -1;
        }

        int lo = 0;
        int hi = nums.length - 1;

        while (lo <= hi) {
            int m = lo + ((hi - lo) >> 1);

            if (nums[m] == target) {
                return m;
            } else if (nums[m] > target) {
                hi = m - 1;
            } else {
                lo = m + 1;
            }
        }

        return -1;
    }

}
