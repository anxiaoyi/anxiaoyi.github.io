    public int findKthLargest(int[] nums, int k) {
        k = nums.length - k;

        // 1 2 3 4 5 6
        //         â†‘(ç¬¬ 2 å¤§)
        //   â†‘(partition = 2 çš„æ—¶å€™ï¼Œå®é™…ä¸ŠæŒ‡å‘çš„æ˜¯è¿™é‡Œ)

        int lo = 0;

        // nums = [1]ï¼Œè¿™ç§æƒ…å†µè¿›å…¥ä¸äº†å¾ªç¯

        while (lo <= hi) {

            if (index == k) {
                return nums[index];
            } else if (index < k) {
                lo = index + 1;
            } else {
                hi = index - 1;
            }
        }

        return -1;
    }

    private int partition(int[] nums, int lo, int high) {
        int i = lo - 1;
        int pivot = nums[high];

        for (int j = lo; j <= high - 1; j++) {
            if (nums[j] <= pivot) {
                i++;
                swap(nums, i, j);
            }
        }

        swap(nums, i + 1, high);
        return i + 1;
    }

    private void swap(int[] array, int i, int j) {
        int temp = array[i];
        array[i] = array[j];
        array[j] = temp;
    }

}
