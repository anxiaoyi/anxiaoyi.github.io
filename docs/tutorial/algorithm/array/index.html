public int findKthLargest(int[] nums, int k) {
    k = nums.length - k;

}

private int partition(int[] nums, int lo, int high) {
    int i = lo - 1;
    int pivot = nums[high];

    for (int j = lo; j <= high - 1; j++) {
        if (nums[j] <= pivot) {
            i++;
            swap(nums, i, j);
        }
    }

    swap(nums, i + 1, high);
    return i + 1;
}

private void swap(int[] array, int i, int j) {
    int temp = array[i];
    array[i] = array[j];
    array[j] = temp;
}

class Solution {
    public int majorityElement(int[] nums) {
        int x = 0, votes = 0;
        
        for (int num : nums){
            if (votes == 0) x = num;
            votes += num == x ? 1 : -1;
        }

        return x;
    }
}
